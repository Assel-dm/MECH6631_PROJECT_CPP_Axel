// bt_robot_controller.ino — Bluetooth HC-06 on hardware Serial (pins 0/1)
// Arduino Uno - Upload via USB first, then disconnect USB and connect HC-06
// HC-06 TX → Pin 0 (RX), HC-06 RX → Pin 1 (TX)

#include <Servo.h>

// Hardware Configuration
const int PIN_MOTOR_LEFT   = 9;
const int PIN_MOTOR_RIGHT  = 10;
const int PIN_LASER_SERVO  = 11;
const int PIN_LASER_ENABLE = 8;

// ⭐ CALIBRATED NEUTRAL VALUES (confirmed working)
const int NEUTRAL_LEFT  = 1500;
const int NEUTRAL_RIGHT = 1480;

Servo servoLeft;
Servo servoRight;
Servo laserServo;

// Safety timeout
const unsigned long TIMEOUT_MS = 500;
unsigned long lastCommandTime = 0;

// Packet buffer
uint8_t packet[6];
int packetIndex = 0;
bool syncFound = false;

void setup() {
  // ⭐ Hardware Serial = HC-06 Bluetooth (pins 0/1)
  // Upload sketch first via USB, then disconnect USB and wire HC-06
  Serial.begin(9600);  // HC-06 default baud rate
  
  // Attach servos
  servoLeft.attach(PIN_MOTOR_LEFT);
  servoRight.attach(PIN_MOTOR_RIGHT);
  laserServo.attach(PIN_LASER_SERVO);
  pinMode(PIN_LASER_ENABLE, OUTPUT);
  
  // Set to calibrated neutral immediately
  servoLeft.writeMicroseconds(NEUTRAL_LEFT);
  servoRight.writeMicroseconds(NEUTRAL_RIGHT);
  laserServo.writeMicroseconds(1000);
  digitalWrite(PIN_LASER_ENABLE, LOW);
  
  lastCommandTime = millis();
}

void loop() {
  // ⭐ STEP 1: Always read Bluetooth packets first
  while (Serial.available() > 0) {
    uint8_t b = Serial.read();
    
    if (b == 0xFF) {
      packetIndex = 0;
      packet[packetIndex++] = b;
      syncFound = true;
    }
    else if (syncFound && packetIndex > 0 && packetIndex < 6) {
      packet[packetIndex++] = b;
      
      if (packetIndex == 6) {
        processCommand();
        lastCommandTime = millis();
        packetIndex = 0;
        syncFound = false;
      }
    }
  }
  
  // ⭐ STEP 2: Timeout AFTER reading - stops if BT disconnects
  if ((millis() - lastCommandTime) > TIMEOUT_MS) {
    stopMotors();
  }
}

void processCommand() {
  uint16_t pw_l = ((uint16_t)packet[1] << 8) | packet[2];
  uint16_t pw_r = ((uint16_t)packet[3] << 8) | packet[4];
  uint8_t  laser = packet[5];
  
  pw_l = constrain(pw_l, 1000, 2000);
  pw_r = constrain(pw_r, 1000, 2000);
  
  servoLeft.writeMicroseconds(pw_l);
  servoRight.writeMicroseconds(pw_r);
  digitalWrite(PIN_LASER_ENABLE, laser ? HIGH : LOW);
  laserServo.writeMicroseconds(laser ? 1500 : 1000);
}

void stopMotors() {
  servoLeft.writeMicroseconds(NEUTRAL_LEFT);
  servoRight.writeMicroseconds(NEUTRAL_RIGHT);
  digitalWrite(PIN_LASER_ENABLE, LOW);
  laserServo.writeMicroseconds(1000);
}
