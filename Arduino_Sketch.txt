#include <Servo.h>

Servo leftMotor;
Servo rightMotor;
Servo laserServo;

// Pin assignments — adjust to your wiring
const int PIN_LEFT  = 9;
const int PIN_RIGHT = 10;
const int PIN_LASER = 11;

// Watchdog: stop motors if no packet received within this many ms
const unsigned long TIMEOUT_MS = 500;
unsigned long last_packet_time = 0;

void setup() {
    Serial.begin(9600);  // Must match BT_BAUD in program.cpp
    leftMotor.attach(PIN_LEFT);
    rightMotor.attach(PIN_RIGHT);
    laserServo.attach(PIN_LASER);

    // Safe stop on startup
    leftMotor.writeMicroseconds(1500);
    rightMotor.writeMicroseconds(1500);
    laserServo.writeMicroseconds(1000);
}

void loop() {
    // Watchdog — stop if PC stops sending
    if (millis() - last_packet_time > TIMEOUT_MS) {
        leftMotor.writeMicroseconds(1500);
        rightMotor.writeMicroseconds(1500);
    }

    // Wait for start byte 0xFF
    if (Serial.available() < 6) return;
    if (Serial.read() != 0xFF) return;  // re-sync on bad byte

    // Wait for remaining 5 bytes
    unsigned long t = millis();
    while (Serial.available() < 5) {
        if (millis() - t > 50) return;  // timeout — drop partial packet
    }

    int pw_l  = (Serial.read() << 8) | Serial.read();
    int pw_r  = (Serial.read() << 8) | Serial.read();
    int laser = Serial.read();

    // Clamp to valid servo range
    pw_l  = constrain(pw_l,  1000, 2000);
    pw_r  = constrain(pw_r,  1000, 2000);

    leftMotor.writeMicroseconds(pw_l);
    rightMotor.writeMicroseconds(pw_r);
    laserServo.writeMicroseconds(laser ? 2000 : 1000);

    last_packet_time = millis();
}
